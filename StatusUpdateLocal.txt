local ReplicatedStorage = game:GetService("ReplicatedStorage")
local voteUpdateEvent = ReplicatedStorage:WaitForChild("VoteUpdate")

local gui = script.Parent
local voteCounter = gui:FindFirstChild("VoteCounter")

voteUpdateEvent.OnClientEvent:Connect(function(votes, totalPlayers)
	voteCounter.Text = votes .. "/" .. totalPlayers
end)
